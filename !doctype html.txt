  <script>
    const W = 640, H = 480;
    const arrowCanvas = document.getElementById('arrowCanvas');
    const ctx = arrowCanvas.getContext('2d');
    const hDirEl = document.getElementById('hDir');
    const vDirEl = document.getElementById('vDir');
    const statusEl = document.getElementById('status');

    function arrow(fromx, fromy, tox, toy, color = '#4CAF50', width = 4) {
      const headlen = 10;
      const angle = Math.atan2(toy - fromy, tox - fromx);
      ctx.strokeStyle = color; ctx.fillStyle = color; ctx.lineWidth = width;
      ctx.beginPath(); ctx.moveTo(fromx, fromy); ctx.lineTo(tox, toy); ctx.stroke();
      ctx.beginPath(); ctx.moveTo(tox, toy);
      ctx.lineTo(tox - headlen * Math.cos(angle - Math.PI/6), toy - headlen * Math.sin(angle - Math.PI/6));
      ctx.lineTo(tox - headlen * Math.cos(angle + Math.PI/6), toy - headlen * Math.sin(angle + Math.PI/6));
      ctx.closePath(); ctx.fill();
    }

    // Convert gaze x,y (in page coordinates) to directions based on thirds of the canvas
    function toDirections(x, y, w, h, deadband = 0.10) {
      // Normalize to 0..1
      const nx = Math.min(Math.max(x / w, 0), 1);
      const ny = Math.min(Math.max(y / h, 0), 1);

      // Deadband around center to avoid flicker
      const leftThresh = 0.5 - deadband, rightThresh = 0.5 + deadband;
      const upThresh = 0.5 - deadband, downThresh = 0.5 + deadband;

      let horizontal = 'CENTER';
      if (nx < leftThresh) horizontal = 'LEFT';
      else if (nx > rightThresh) horizontal = 'RIGHT';

      let vertical = 'CENTER';
      if (ny < upThresh) vertical = 'UP';
      else if (ny > downThresh) vertical = 'DOWN';

      return { horizontal, vertical, nx, ny };
    }

    function draw(nx, ny) {
      ctx.clearRect(0, 0, W, H);
      // Start from center
      const cx = W/2, cy = H/2;
      // Target toward gaze point
      const tx = nx * W, ty = ny * H;
      arrow(cx, cy, tx, ty, '#4CAF50', 4);
      // Dot at target
      ctx.fillStyle = '#4CAF50';
      ctx.beginPath(); ctx.arc(tx, ty, 5, 0, Math.PI*2); ctx.fill();
    }

    async function start() {
      statusEl.textContent = 'Initializing WebGazer...';
      // Optional UI
      webgazer.showVideoPreview(true)     // video feed
              .showPredictionPoints(false)
              .showFaceOverlay(true)
              .showFaceFeedbackBox(true);

      await webgazer.setRegression('ridge')   // default is fine
                    .begin();

      statusEl.textContent = 'Running. Look around.';

      // Main loop
      function tick() {
        const pred = webgazer.getCurrentPrediction();
        if (pred && Number.isFinite(pred.x) && Number.isFinite(pred.y)) {
          // Map page coords to canvas coords (assuming canvas is 640x480 and positioned normally)
          const rect = arrowCanvas.getBoundingClientRect();
          const x = pred.x - rect.left;  // local to canvas
          const y = pred.y - rect.top;

          const { horizontal, vertical, nx, ny } = toDirections(x, y, rect.width, rect.height, 0.12);

          hDirEl.textContent = horizontal;
          vDirEl.textContent = vertical;

          draw(nx, ny);
        }
        requestAnimationFrame(tick);
      }
      requestAnimationFrame(tick);
    }

    window.addEventListener('load', start);
  </script>
